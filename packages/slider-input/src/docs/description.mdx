## Стандартный вид компонента

Можно настроить количество сегментов и шаг внутри сегмента.

```jsx live
const min = 1;

render(() => {
    const [value1, setValue1] = React.useState(3);
    const [value2, setValue2] = React.useState(5);

    return (
        <div style={{ width: isMobile() ? '100%' : 320 }}>
            <SliderInput
                value={value1}
                label='Label'
                labelView={isMobile() ? 'outer' : 'inner'}
                size={isMobile() ? 's' : 'm'}
                pips={{
                    mode: 'values',
                    values: [min, 10 / 2, 10],
                }}
                min={min}
                max={10}
                step={1}
                block={true}
                onChange={(_, { value }) => setValue1(value)}
                breakpoint={BREAKPOINT}
            />
            <Gap size='xl' />
            <SliderInput
                value={value2}
                label='Label'
                labelView={isMobile() ? 'outer' : 'inner'}
                size={isMobile() ? 's' : 'm'}
                pips={{
                    mode: 'values',
                    values: [min, 10, 100],
                }}
                min={min}
                max={100}
                range={{
                    min: min,
                    '50%': [10, 5],
                    max: 100,
                }}
                step={1}
                block={true}
                onChange={(_, { value }) => setValue2(value)}
                breakpoint={BREAKPOINT}
            />
        </div>
    );
});
```

## Состояния

В состоянии ошибки шкала заменяется на сообщение об ошибке, а в задисейбленном состоянии может превращаться в подпись.

```jsx live
const formatAmountSimple = (() => {
    const formatter = new Intl.NumberFormat('ru-RU', {
        style: 'currency',
        currency: 'RUB',
    });

    return (value) => (value ? formatter.format(value).slice(0, -5) : '');
})();

const min = 100000;
const max = 300000;
const pips = {
    mode: 'values',
    values: [min, max - min, max],
    format: {
        to: (value) => formatAmountSimple(value),
    },
};

render(() => {
    const [value, setValue] = React.useState(150000);
    const [props, setProps] = React.useState({});

    return (
        <div style={{ width: '100%' }}>
            <div style={{ width: isMobile() ? '100%' : 320 }}>
                <SliderInput
                    value={formatAmountSimple(value)}
                    sliderValue={value}
                    label='Label'
                    labelView={isMobile() ? 'outer' : 'inner'}
                    size={isMobile() ? 's' : 'm'}
                    pips={pips}
                    min={min}
                    max={max}
                    step={1000}
                    block={true}
                    onChange={(_, { value }) => setValue(value)}
                    breakpoint={BREAKPOINT}
                    {...props}
                />
            </div>
            <Gap size='xl' />

            <Space direction='horizontal' wrap={true}>
                <Button onClick={() => setProps({})} size='xxs' view='secondary'>
                    По умолчанию
                </Button>
                <Button onClick={() => setProps({ error: 'Error' })} size='xxs' view='secondary'>
                    Ошибка
                </Button>
                <Button
                    onClick={() => setProps({ disabled: true, hint: 'Hint' })}
                    size='xxs'
                    view='secondary'
                >
                    Заблокирован
                </Button>
            </Space>
        </div>
    );
});
```

## Amount

Используйте Amount для деления введенных значений на разряды.

```jsx live
const min = 50000;
const max = 10000000;
const step = 1000;
const range = {
    min: [min],
    '25%': [100000, 10000],
    '75%': [1000000, 100000],
    max: [max],
};
const pips = {
    mode: 'values',
    values: [min, max],
    format: {
        to: (value) => {
            if (value < 1000000) return `${value / 1000} тыс`;
            return `${value / 1000000} млн`;
        },
    },
};

render(() => {
    const [value, setValue] = React.useState(120000);

    const handleInputChange = (event, { value }) => {
        setValue(value !== '' ? (value / 100).toString() : '');
    };

    const handleSliderChange = ({ value }) => {
        setValue(value);
    };

    const handleBlur = () => {
        if (value !== '') {
            setValue(Math.max(min, Math.min(max, parseFloat(value.toString()))));
        }
    };

    return (
        <div style={{ width: isMobile() ? '100%' : 320 }}>
            <SliderInput
                block={true}
                value={value !== '' ? value * 100 : ''}
                sliderValue={value}
                onInputChange={handleInputChange}
                onSliderChange={handleSliderChange}
                onBlur={handleBlur}
                min={min}
                max={max}
                range={range}
                pips={pips}
                step={step}
                label='Первоначальный взнос'
                Input={AmountInput}
                labelView={isMobile() ? 'outer' : 'inner'}
                size={isMobile() ? 's' : 'm'}
                breakpoint={BREAKPOINT}
            />
        </div>
    );
});
```
