// import { withSuffix } from '@alfalab/core-components/with-suffix';
// import { Input } from '@alfalab/core-components/input';

const SuffixInput = withSuffix(Input);

const getSuffix = value => {
    if (value <= 0) return '';

    if (value.toString().endsWith('1')) {
        return ' год';
    }

    if (
        value.toString().endsWith('2') ||
        value.toString().endsWith('3') ||
        value.toString().endsWith('4')
    ) {
        return ' года';
    }

    return ' лет';
};

const min = 1;
const max = 5;
const step = 1;
const pips = {
    mode: 'values',
    values: [1, 2, 3, 4, 5],
    format: {
        to: value => `${value}${getSuffix(value)}`,
    },
};

const normalizeValue = (value, min, max) =>
    value === '' ? '' : Math.min(max, value.toString().replace(/[^\d-]/g, ''));

render(() => {
    const [value, setValue] = React.useState('');

    const handleChange = (event, { value }) => setValue(normalizeValue(value, min, max));

    return (
        <SliderInput
            value={value}
            min={min}
            max={max}
            step={step}
            onChange={handleChange}
            block={true}
            label='Срок кредита'
            pips={pips}
            Input={SuffixInput}
            customInputProps={{
                suffix: getSuffix(value),
            }}
        />
    );
});

## AmountSliderInput

```jsx live
// import { AmountInput } from '@alfalab/core-components/amount-input';

const min = 100000;
const max = 600000;
const step = 1000;
const pips = {
    mode: 'values',
    values: [100000, 600000],
    format: {
        to: value => `${Math.floor(value / 1000)} тыс.`,
    },
};

render(() => {
    const [value, setValue] = React.useState('');

    const handleInputChange = (event, { value }) => {
        setValue(value !== '' ? (+value / 100).toString() : '');
    };

    const handleSliderChange = ({ value }) => {
        setValue(value);
    };

    const handleBlur = () => {
        if (value !== '') {
            setValue(Math.max(min, Math.min(max, +value)));
        }
    };

    return (
        <SliderInput
            value={value !== '' ? value * 100 : ''}
            sliderValue={value}
            onInputChange={handleInputChange}
            onSliderChange={handleSliderChange}
            onBlur={handleBlur}
            min={min}
            max={max}
            step={step}
            Input={AmountInput}
            pips={pips}
            label='Сумма кредита'
        />
    );
});
```
