render(() => {
    const [state, setState] = React.useState({ stiffness: 100, damping: 10, mass: 1 });

    return (
        <>
            <div style={{ padding: '16px' }}>
                <ButtonDesktop view='primary'>
                    Accent
                </ButtonDesktop>
                <Gap size={24} />
                <div style={{ marginBottom: '16px', display: 'flex', gap: '8px', flexWrap: 'wrap' }}>
                    <div style={{ minWidth: '200px' }}>
                        <div style={{ fontSize: '12px', marginBottom: '4px' }}>Stiffness: {state.stiffness}</div>
                        <Gap size={8} />
                        <Slider
                            value={state.stiffness}
                            min={1}
                            max={300}
                            step={1}
                            onChange={({ value }) => setState(s => ({ ...s, stiffness: value }))}
                        />
                    </div>
                    <div style={{ minWidth: '200px' }}>
                        <div style={{ fontSize: '12px', marginBottom: '4px' }}>Damping: {state.damping}</div>
                        <Gap size={8} />
                        <Slider
                            value={state.damping}
                            min={1}
                            max={50}
                            step={1}
                            onChange={({ value }) => setState(s => ({ ...s, damping: value }))}
                        />
                    </div>
                    <div style={{ minWidth: '200px' }}>
                        <div style={{ fontSize: '12px', marginBottom: '4px' }}>Mass: {state.mass}</div>
                        <Gap size={8} />
                        <Slider
                            value={state.mass}
                            min={0.1}
                            max={5}
                            step={0.1}
                            onChange={({ value }) => setState(s => ({ ...s, mass: value }))}
                        />
                    </div>
                </div>
            </div>
        </>
    );
});
