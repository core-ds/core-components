import { StarMIcon } from '@alfalab/icons-glyph/StarMIcon';
import { StarSIcon } from '@alfalab/icons-glyph/StarSIcon';

export const VIEWS = ['accent', 'primary', 'secondary', 'tertiary', 'link', 'ghost'];
export const SIZES = ['xxs', 'xs', 's', 'm', 'l', 'xl'];

<Meta title='Components/Button' component={Button} id='Button' />

{/* Canvas */}

<Story name='Button'>
    {React.createElement(() => {
        const colors = select('colors', ['default', 'inverted'], 'default');
        const sizel = select('size', SIZES, 'm');
        const addons = sizel === 'xxs' ? <StarSIcon /> : <StarMIcon />;
        return (
            <div
                style={{
                    backgroundColor:
                        colors === 'inverted'
                            ? 'var(--color-light-bg-primary-inverted)'
                            : 'transparent',
                    padding: '8px',
                    position: 'absolute',
                    top: 0,
                    left: 0,
                    right: 0,
                    bottom: 0,
                }}
            >
                <Button
                    view={select('view', VIEWS, 'primary')}
                    colors={colors}
                    size={sizel}
                    href={text('href', '')}
                    loading={boolean('loading')}
                    disabled={boolean('disabled', false)}
                    block={boolean('block', false)}
                    nowrap={boolean('nowrap', false)}
                    leftAddons={boolean('leftAddons', false) && addons}
                    rightAddons={boolean('rightAddons', false) && addons}
                >
                    {text('label', 'Оплатить')}
                </Button>
            </div>
        );
    })}
</Story>
