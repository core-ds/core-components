```jsx
import { ContentDesktop, ContentMobile } from '@alfalab/core-components/universal-modal';
import type { ContentDesktopProps, ContentMobileProps } from '@alfalab/core-components/universal-modal';

import { Controls } from '@alfalab/core-components/universal-modal';
import type { ControlsProps } from '@alfalab/core-components/universal-modal';

import { HeaderResponsive, HeaderDesktop, HeaderMobile } from '@alfalab/core-components/universal-modal';

import { FooterDesktop, FooterMobile } from '@alfalab/core-components/universal-modal';
import type { FooterDesktopProps, FooterMobileProps } from '@alfalab/core-components/universal-modal';
```

Из индекса импортируется responsive версия компонента.

## Использование dataTestId

В компоненте используются модификаторы для `dataTestId`.
Для удобного поиска элементов можно воспользоваться функцией `getUniversalModalTestIds`.
Импорт из `@alfalab/core-components/universal-modal/shared`.

Функция возвращает объект:

```jsx
{
        modal: dataTestId,
        header: `${dataTestId}-header`,
        title: `${dataTestId}-header-title`,
        content: `${dataTestId}-content`,
        footer: `${dataTestId}-footer`,
};
```
