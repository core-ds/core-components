import { Meta, Story, Markdown } from '@storybook/addon-docs';
import { boolean, select, text } from '@storybook/addon-knobs';
import { ComponentHeader, Tabs } from 'storybook/blocks';

import { Button } from '@alfalab/core-components-button';
import { ButtonDesktop } from '@alfalab/core-components-button/desktop';
import { ButtonMobile } from '@alfalab/core-components-button/mobile';
import { UniversalModalMobile } from '@alfalab/core-components-universal-modal/mobile';
import { UniversalModalDesktop } from '@alfalab/core-components-universal-modal/desktop';
import { UniversalModalResponsive } from '@alfalab/core-components-universal-modal/Component.responsive';

import Description from './description.mdx';
import Development from './development.mdx';
import Changelog from '../../CHANGELOG.md?raw';

<Meta title='Components/UniversalModal' id='UniversalModal' />

export const Text = () => (
    <p style={{ margin: '0 0 16px' }}>
        Сейчас много говорят об отказах банков в проведении операций, блокировках интернет-банка.
        Это связано с тем, что Центральный банк РФ обязывает банки выявлять операции своих клиентов,
        потенциально нарушающие требования Федерального закона «О противодействии легализации
        (отмыванию) доходов, полученных преступным путем, и финансированию терроризма» — 115-ФЗ
    </p>
);

export const Content = () => {
    const [showMore, setShowMore] = React.useState(false);
    return (
        <>
            <Text />
            {(showMore || boolean('showMore')) && (
                <>
                    <Text />
                    <Text />
                    <Text />
                    <Text />
                    <Text />
                </>
            )}
            <Button size='xs' type='Button' onClick={() => setShowMore(!showMore)}>
                {showMore ? 'Скрыть' : 'Показать еще'}
            </Button>
        </>
    );
};

<Story name='UniversalModal'>
    {React.createElement(() => {
        const [open, setOpen] = React.useState(false);
        const handleModalOpen = () => setOpen(!open);

        const component = select(
            'Component',
            ['UniversalModalResponsive', 'UniversalModalDesktop', 'UniversalModalMobile'],
            'UniversalModalDesktop',
        );
        const horizontalAlign = select('horizontalAlign', ['left', 'center', 'right'], 'center');
        const verticalAlign = select('verticalAlign', ['top', 'center', 'bottom'], 'center');
        const width = text('width', '500');
        const height = text('height', '500');
        const overlay = boolean('overlay', true);
        const header = boolean('header', true);
        const headerTitle = text('header.title');
        const headerAlign = select('header.align', ['left', 'center'], 'left');
        const stickyHeader = boolean('header.sticky', false);
        const stickyFooter = boolean('footer.sticky', false);
        const footerLayout = select(
            'footer.layout',
            ['start', 'center', 'space-between', 'column'],
            'start',
        );

        const Component = React.useMemo(() => {
            return {
                UniversalModalResponsive,
                UniversalModalDesktop,
                UniversalModalMobile,
            }[component];
        }, [component]);

        const isMobile = Component === UniversalModalMobile;
        const FooterButton = isMobile ? ButtonMobile : ButtonDesktop;

        return (
            <div style={{ display: 'inline-block', background: 'transparent'}}>
                <Button type='button' size='xs' onClick={handleModalOpen}>
                    Открыть модалку
                </Button>
                <Component
                    open={open || boolean('open')}
                    onClose={handleModalOpen}
                    horizontalAlign={horizontalAlign}
                    verticalAlign={verticalAlign}
                    width={width}
                    height={height}
                    overlay={overlay}
                >
                    {header && (
                        <Component.Header
                            sticky={stickyHeader}
                            title={headerTitle}
                            align={headerAlign}
                            bigTitle={boolean('bigTitle', false)}
                        />
                    )}
                    <Component.Content>
                        <Content />
                    </Component.Content>

                    {stickyFooter && (
                        <Component.Footer sticky={stickyFooter}>
                            <Component.Controls
                                layout={footerLayout}
                                primary={
                                    <FooterButton size={isMobile ? 'm' : 's'} view='primary'>
                                        Сохранить
                                    </FooterButton>
                                }
                                secondary={
                                    <FooterButton size={isMobile ? 'm' : 's'} view='secondary'>
                                        Отмена
                                    </FooterButton>
                                }
                            />
                        </Component.Footer>
                    )}
                </Component>
            </div>
        );
    })}

</Story>

{/* Docs */}

<ComponentHeader
    name='UniversalModal'
    children='Модальное окно. Используется для отображения контента в модальном слое.'
/>

<Tabs
    description={<Description />}
    development={<Development />}
    changelog={<Markdown>{Changelog}</Markdown>}
/>
