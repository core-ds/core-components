## Примеры наполнения

В десктопных модалках в зависимости от положения и размера меняется стиль заголовка в хедере и размер кнопок в футере.

```jsx live mobileHeight={640}
const horizontalOptions = [
    { key: 'start', content: 'Start' },
    { key: 'center', content: 'Center' },
    { key: 'end', content: 'End' },
];
const verticalOptions = [
    { key: 'top', content: 'Top' },
    { key: 'center', content: 'Center' },
    { key: 'bottom', content: 'Bottom' },
];

const heightOptions = [
    { key: 'hugContent', content: 'Hug Content' },
    { key: '500', content: '500' },
    { key: '600', content: '600' },
    { key: '800', content: '800' },
    { key: 'fullHeight', content: 'Full' },
];

const widthOptions = [
    { key: '500', content: '500' },
    { key: '600', content: '600' },
    { key: '800', content: '800' },
    { key: 'fullWidth', content: 'Full' },
];

render(() => {
    const [open, setOpen] = React.useState(false);
    const [horizontalAlign, setHorizontalAlign] = React.useState('center');
    const [verticalAlign, setVerticalAlign] = React.useState('center');
    const [modalWidth, setModalWidth] = React.useState('500');
    const [modalHeight, setModalHeight] = React.useState('500');

    const [headerButtonCross, setHeaderButtonCross] = React.useState(true);
    const [headerButtonArrow, setHeaderButtonArrow] = React.useState(false);
    const [headerType, setHeaderType] = React.useState('none');

    const [mainFooterButton, setMainFooterButton] = React.useState(true);
    const [additionalFooterButton, setAdditionalFooterButton] = React.useState(false);
    const [verticalFooterButtons, setVerticalFooterButtons] = React.useState(false);
    const [stretchFooterButtons, setStretchFooterButtons] = React.useState(false);

    const [contentTitle, setContentTitle] = React.useState(false);
    const [contentButton, setContentButton] = React.useState(false);

    const handleRadioChange = (state) => {
        setHeaderType(state);
    };

    const getButtonSize = () => {
        if (horizontalAlign !== 'center') {
            return 56;
        } else {
            if (modalWidth >= 800) {
                return 56;
            }
            return 48;
        }
    };

    const getMargin = () => {
        if (horizontalAlign === 'center') {
            if (verticalAlign === 'top') {
                return {
                    top: 12,
                    ...(modalWidth === 'fullWidth' && {
                        right: 12,
                        left: 12,
                    }),
                    ...(modalHeight === 'fullHeight' && {
                        bottom: 12,
                    }),
                };
            }

            if (verticalAlign === 'center') {
                return {
                    ...(modalWidth === 'fullWidth' && {
                        right: 12,
                        left: 12,
                    }),
                    ...(modalHeight === 'fullHeight' && {
                        top: 12,
                        bottom: 12,
                    }),
                };
            }

            if (verticalAlign === 'bottom') {
                return {
                    bottom: 12,
                    ...(modalWidth === 'fullWidth' && {
                        right: 12,
                        left: 12,
                    }),
                    ...(modalHeight === 'fullHeight' && {
                        top: 12,
                    }),
                };
            }
        } else {
            return { top: 12, right: 12, bottom: 12, left: 12 };
        }
    };

    return (
        <React.Fragment>
            <Button onClick={() => setOpen((prev) => !prev)}>Показать модалку</Button>
            <Gap size={20} />
            <Divider />
            <Gap size={20} />
            <div style={{ display: 'flex', flexDirection: 'column', width: 532 }}>
                <div style={{ display: 'flex' }}>
                    <div style={{ width: 260 }}>
                        <SelectDesktop
                            size={56}
                            options={horizontalOptions}
                            selected={horizontalAlign}
                            label='Положение по горизонтали'
                            block={true}
                            onChange={({ selected }) => setHorizontalAlign(selected.key)}
                        />
                    </div>
                    <Gap size={12} direction={'horizontal'} />
                    <div style={{ width: 260 }}>
                        <SelectDesktop
                            size={56}
                            options={verticalOptions}
                            selected={verticalAlign}
                            label='Положение по вертикали'
                            block={true}
                            onChange={({ selected }) => setVerticalAlign(selected.key)}
                        />
                    </div>
                </div>
                <Gap size={12} />
                <div style={{ display: 'flex' }}>
                    <div style={{ width: 260 }}>
                        <SelectDesktop
                            size={56}
                            options={heightOptions}
                            selected={modalHeight}
                            label='Высота'
                            block={true}
                            onChange={({ selected }) => setModalHeight(selected.key)}
                        />
                    </div>
                    <Gap size={12} direction={'horizontal'} />
                    <div style={{ width: 260 }}>
                        <SelectDesktop
                            size={56}
                            options={widthOptions}
                            selected={modalWidth}
                            label='Ширина'
                            block={true}
                            onChange={({ selected }) => setModalWidth(selected.key)}
                        />
                    </div>
                </div>
            </div>
            <Gap size={24} />
            Навигация
            <Gap size={12} />
            <div>
                <Switch
                    label='Крестик'
                    checked={headerButtonCross}
                    onChange={() => setHeaderButtonCross((prev) => !prev)}
                />
                <Gap size={12} />
                <Switch
                    label='Стрелка назад'
                    checked={headerButtonArrow}
                    onChange={() => setHeaderButtonArrow((prev) => !prev)}
                />
                <Gap size={12} />
                Заголовок в шапке
                <Gap size={12} />
                <div style={{ display: 'flex', flexDirection: 'column' }}>
                    <Radio
                        size={24}
                        onChange={() => handleRadioChange('normal')}
                        checked={headerType === 'normal'}
                        label='Обычный'
                        hint='Для модалок шириной до 800 рх'
                    />
                    <Gap size={12} />
                    <Radio
                        size={24}
                        onChange={() => handleRadioChange('big')}
                        checked={headerType === 'big'}
                        label='Крупный'
                        hint='Для широких модалок и сайдпанелей'
                    />
                    <Gap size={12} />
                    <Radio
                        size={24}
                        onChange={() => handleRadioChange('none')}
                        checked={headerType === 'none'}
                        label='Без заголовка'
                    />
                </div>
            </div>
            <Gap size={24} />
            Настройки футера
            <Gap size={12} />
            <div>
                <Switch
                    label='Основная кнопка'
                    checked={mainFooterButton}
                    onChange={() => {
                        setContentButton(false);
                        setMainFooterButton((prev) => !prev);
                    }}
                />
                <Gap size={12} />
                <Switch
                    label='Дополнительная кнопка'
                    checked={additionalFooterButton}
                    onChange={() => {
                        setContentButton(false);
                        setAdditionalFooterButton((prev) => !prev);
                    }}
                />
                <Gap size={12} />
                <Switch
                    label='Вертикальная ориентация'
                    checked={verticalFooterButtons}
                    onChange={() => setVerticalFooterButtons((prev) => !prev)}
                />
                <Gap size={12} />
                <Switch
                    label='Растянуть кнопки на всю ширину'
                    hint='Только для горизонтального расположения'
                    checked={stretchFooterButtons}
                    onChange={() => setStretchFooterButtons((prev) => !prev)}
                />
            </div>
            <Gap size={24} />
            Настройки контента
            <Gap size={12} />
            <div>
                <Switch
                    label='Заголовок'
                    hint='Если не стали размещать его в шапке'
                    checked={contentTitle}
                    onChange={() => {
                        setContentTitle((prev) => !prev);
                    }}
                />
                <Gap size={12} />
                <Switch
                    label='Кнопка'
                    hint='Если не стали размещать его в футере'
                    checked={contentButton}
                    onChange={() => {
                        setMainFooterButton(false);
                        setAdditionalFooterButton(false);
                        setContentButton((prev) => !prev);
                    }}
                />
            </div>
            <UniversalModalDesktop
                open={open}
                width={modalWidth}
                height={modalHeight}
                horizontalAlign={horizontalAlign}
                verticalAlign={verticalAlign}
                margin={getMargin()}
                onClose={() => setOpen(false)}
            >
                <UniversalModalDesktop.Header
                    sticky={true}
                    {...(headerType !== 'none' &&
                        !headerButtonArrow && {
                            align: 'left',
                            title: 'Почему банк проверяет мои операции?',
                            bigTitle: headerType === 'big',
                        })}
                    {...(headerButtonArrow && {
                        hasBackButton: headerButtonArrow,
                        onBack: () => setOpen(false),
                        align: 'center',
                        ...(headerType !== 'none' && {
                            bottomAddons: 'Почему банк проверяет мои операции?',
                            bigTitle: headerType === 'big',
                        }),
                    })}
                    hasCloser={headerButtonCross}
                />
                <UniversalModalDesktop.Content>
                    {contentTitle && (
                        <React.Fragment>
                            <Typography.Title view='medium' font='system'>
                                Почему банк проверяет мои операции?
                            </Typography.Title>
                            <Gap size={16} />
                        </React.Fragment>
                    )}

                    <Typography.Text tag='p'>
                        В 2001 году в России начал действовать Федеральный закон №115 «О
                        противодействии легализации доходов, полученных преступным путём, и
                        финансированию терроризма». В рамках закона банки могут блокировать карты,
                        отказывать в проведении сомнительных операций, ограничить доступ в

                        «антиотмывочного» законодательства всегда неприятны: приходится остановить
                        бизнес-процессы и доказать банку законность операций. Специалисты
                        «Альфа-банка» собрали понятные рекомендации, как сэкономить время на
                        объяснения и предотвратить блокировки
                    </Typography.Text>

                    <Collapse collapsedLabel='Подробнее' expandedLabel='Скрыть'>
                        <Typography.Text tag='p'>
                            115-ФЗ Касается всех предпринимателей, фирм и физлиц, а также тех, кто

                            личных расчётов. Ограничения интернет-банка, блокировка карт
                            добросовестных компаний могут произойти из-за неправильно оформленных
                            документов, ошибок в платёжке или попыток снизить налоги.
                        </Typography.Text>
                        <br />
                        <Typography.Text tag='p'>
                            Клиенты воспринимают ограничения как атаку со стороны банка, но чаще
                            всего сами допускают ошибки или нарушения, которых можно избежать. Банки
                            не преследуют цели доставить неудобства клиентам — они обязаны соблюдать
                            законодательство и следовать инструкциям и рекомендациям ЦБ, а в
                            противном случае рискуют лишиться лицензии.
                        </Typography.Text>
                        <br />
                        <Typography.Text tag='p'>
                            Обналичивание — сомнительные операции, когда юрлицо или предприниматель
                            снимает со счёта более 80% от оборота или переводит деньги на счета
                            физлиц, которые затем снимают в наличной форме.
                        </Typography.Text>
                        <br />
                        <Typography.Text tag='p'>
                            Вывод капитала за границу — это переводы нерезидентам по договорам об
                            импорте работ/услуг и результатов интеллектуальной деятельности, по
                            которым проведение расчётов осуществляется без одновременной уплаты НДС;
                            по сделкам купли-продажи ценных бумаг, а также товаров, которые не
                            пересекают границу России.
                        </Typography.Text>
                        <br />
                        <Typography.Text tag='p'>
                            Транзитные операции — операции, в процессе которых деньги поступают на
                            счёт компании от других резидентов и списываются в короткие сроки. При
                            этом, как правило, в этих случаях по счёту нет начислений зарплат,
                            уплаты налогов, и они не соответствуют заявленному компанией виду
                            деятельности.
                        </Typography.Text>
                        <br />
                        <Typography.Text tag='p'>
                            Запрашивать могут любые документы и устанавливать разные сроки их
                            предоставления — это зависит от службы контроля конкретного банка.
                            Обычно банки запрашивают чеки, счета или договора с контрагентами. В

                            банк для устного разъяснения или выехать по месту ведения бизнеса
                            клиента.
                        </Typography.Text>
                    </Collapse>
                    {contentButton && (
                        <React.Fragment>
                            <Gap size={16} />
                            <Gap size={24} />
                            <Button size={48} onClick={() => setOpen(false)}>
                                Кнопка
                            </Button>
                        </React.Fragment>
                    )}
                </UniversalModalDesktop.Content>
                {(mainFooterButton || additionalFooterButton) && (
                    <UniversalModalDesktop.Footer
                        sticky={true}
                        layout={verticalFooterButtons ? 'column' : 'start'}

                                view='primary'

                                onClick={() => setOpen(false)}
                            >
                                Primary
                            </Button>
                        )}

                        {additionalFooterButton && (
                            <Button
                                size={getButtonSize()}
                                view='secondary'
                                block={stretchFooterButtons}
                                onClick={() => setOpen(false)}
                            >
                                Secondary
                            </Button>
                        )}
                    </UniversalModalDesktop.Footer>
                )}
            </UniversalModalDesktop>
        </React.Fragment>
    );
});
```

## Смена модалок

При смене одной модалки на другую важно дождаться окончания анимации закрытия первой модалки перед появлением новой.
