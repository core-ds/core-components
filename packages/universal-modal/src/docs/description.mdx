## Анатомия

Modal построен как Compound Components. Состоит из компонентов Header, Content и Footer, содержащих в себе кастомизируемые слоты.

```jsx live mobileHeight={640}
render(() => {
    const [open, setOpen] = React.useState(false);

    const dashedBackground = {
        boxSizing: 'border-box',
        border: '1px dashed #2288FA',
        backgroundColor: '#D8EAFF',
        borderRadius: 8,
    };

    const headerAnatomyStyle = {
        padding: '20px 20px 0',
        display: 'flex',
        addonStyles: {
            width: 48,
            height: 48,
            ...dashedBackground,
        },
        titleStyles: {
            flexGrow: 1,
            height: 48,
            ...dashedBackground,
        },
    };

    const contentAnatomyStyles = {
        padding: '0 32px',
        height: 336,
        content: {
            height: '100%',
            ...dashedBackground,
        },
    };

    const footerAnatomyStyles = {
        boxSizing: 'border-box',
        padding: '16px 32px 32px',
        footer: {
            height: 48,
            ...dashedBackground,
        },
    };

    return (
        <React.Fragment>
            <Button onClick={() => setOpen((prev) => !prev)}>Показать анатомию</Button>
            <UniversalModalResponsive
                open={open}
                width={500}
                height={500}
                onClose={() => setOpen(false)}
            >
                <div style={{ ...headerAnatomyStyle }}>
                    <div style={{ ...headerAnatomyStyle.addonStyles }} />
                    <div style={{ ...headerAnatomyStyle.titleStyles }} />
                    <div style={{ ...headerAnatomyStyle.addonStyles }} />
                </div>
                <div style={{ ...contentAnatomyStyles }}>
                    <div style={{ ...contentAnatomyStyles.content }}></div>
                </div>
                <div style={{ ...footerAnatomyStyles }}>
                    <div style={{ ...footerAnatomyStyles.footer }}></div>
                </div>
            </UniversalModalResponsive>
        </React.Fragment>
    );
});
```
