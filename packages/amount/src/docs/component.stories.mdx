<Story name='Amount'>

        const value = number('value', 10099);

        const showPlus = boolean('showPlus', false);
        const view = select('view', ['default', 'withZeroMinorPart'], 'default');

        const transparentMinor = boolean('transparentMinor', true);
        return (
            <Container>
                <Row>
                    <Col>Amount</Col>
                    <Col>Amount.Pure (Без стилей)</Col>
                </Row>
                <Row>
                    <Col>
                        <Amount
                            value={value}
                            currency={currency}
                            minority={minority}
                            showPlus={showPlus}
                            view={view}
                            bold={bold}
                            transparentMinor={transparentMinor}
                        />
                    </Col>
                    <Col>
                        <Amount.Pure
                            value={value}
                            currency={currency}
                            minority={minority}
                            showPlus={showPlus}
                            view={view}
                        />
                    </Col>

    })}
</Story>
