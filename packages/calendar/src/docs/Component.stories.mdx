<Story name='Calendar'>
    {React.createElement(() => {
        const offDays = [new Date().setDate(7)];
        const events = [
            new Date().setDate(3),
            new Date().setDate(17),
        ];
        const [open, setOpen] = React.useState(false);
        const [value, setValue] = React.useState();
        const { selectedFrom, selectedTo, updatePeriod } = usePeriod();
        const device = select('device', ['desktop', 'mobile'], 'desktop');
        const selectorView = select('selectorView', ['full', 'month-only'], 'full');
        const defaultView = select('defaultView', ['days', 'months', 'years'], 'days');
        const renderContent = () => {
            if (device === 'desktop') {
                return (
                    <Calendar
                        value={value}
                        onChange={setValue}
                        offDays={offDays}
                        events={events}
                        defaultView={defaultView}
                        selectorView={selectorView}
                    />
                )
            }
            else if (selectorView === 'full') {
                return (
                    <>
                        <Button onClick={() => setOpen(true)}>
                            Открыть календарь
                        </Button>
                        <CalendarMobile
                            onClose={() => setOpen(false)}
                            open={open}
                            value={value}
                            onChange={setValue}
                            defaultView={defaultView}
                            selectorView={selectorView}
                        />
                    </>
                )
            }
            else {
                return (
                    <>
                        <Button onClick={() => setOpen(true)}>
                            Открыть календарь (период)
                        </Button>
                        <CalendarMobile
                            onClose={() => setOpen(false)}
                            open={open}
                            onChange={updatePeriod}
                            selectedFrom={selectedFrom}
                            selectedTo={selectedTo}
                            defaultView={defaultView}
                            selectorView={selectorView}
                        />
                    </>
                )
            }
        }
        return renderContent();
    })}
</Story>
