Компонент текстового ввода даты с указанием времени

## Примеры

### Без пикера
```jsx live
render(<DateTimeInput label="Укажите дату и время" placeholder="ДД.ММ.ГГГГ, ЧЧ:ММ" />)
```

### С пикером
```jsx live
render(<DateTimeInput label="Укажите дату и время" placeholder="ДД.ММ.ГГГГ, ЧЧ:ММ" picker />)
```

### С обработчиками событий
```jsx live
render(() => {
    const handleChange = (event, payload) => {
        const { date, value } = payload;
        console.log("Changed", date, value)
    }

    const handleComplete = (event, payload) => {
        const { date, value } = payload;
        console.log("Completed", date, value)
    }

    return (<DateTimeInput onChange={handleChange} onComplete={handleComplete} label="Укажите диапазон дат" placeholder="ДД.ММ.ГГГГ, ЧЧ:ММ" picker />)
})
```
