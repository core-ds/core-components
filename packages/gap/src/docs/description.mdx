## Примеры

```jsx live
render(() => {
    const [direction, setDirection] = React.useState('horizontal');

    const stylesAddon={
        width: '64px',
        height: '64px',
        borderRadius: '8px',
        backgroundColor: 'rgba(38, 55, 88, 0.06)',
        boxSizing: 'border-box',
    }
    
    const stylesWrapper = {
        display: 'flex',
        flexDirection: direction === 'vertical' ? 'column' : 'row',
    }

    return (
        <div>
            <div style={stylesWrapper}>
                <div style={stylesAddon}/>
                <Gap size='m' direction={direction} />
                <div style={stylesAddon}/>
                <Gap size='m' direction={direction} />
                <div style={stylesAddon}/>
            </div>
            <Gap size='xl' />
            <RadioGroup
                label='Лейаут'
                direction='vertical'
                name='radioGroupFirst'
                onChange={(_, { value }) => setDirection(value)}
                value={direction}
            >
                <Radio label='Горизонтальный' value='horizontal' />
                <Radio label='Вертикальный' value='vertical' />
            </RadioGroup>
        </div>
    );
});
```
