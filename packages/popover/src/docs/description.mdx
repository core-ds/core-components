## Пример

Под капотом использует floating-ui и react-transition-group.

```jsx live mobileWidth={400}
render(() => {
    const [count, setCount] = React.useState(0);
    const buttonElementRef = React.useRef(null);

    const label = {
        0: 'Показать поповер снизу',
        1: 'Показать поповер сверху',
        2: 'Показать слева',
        3: 'Показать справа',
        4: 'Скрыть',
    };

    const position = {
        1: 'bottom',
        2: 'top',
        3: 'left',
        4: 'right',
    };

    const onClick = () => {
        setCount((count) => count === 4 ? 0 : count + 1);
    };
    const popoverChildren = (
        <>{count !== 0 && <div style={{ padding: '15px', minWidth: '156px' }}>Popover</div>}</>
    );

    return (
        <div
            style={{
                width: '700px',
                height: '300px',
                display: 'flex',
                justifyContent: 'center',
                alignItems: 'center',
            }}
        >
            <Popover
                anchorElement={buttonElementRef}
                useAnchorWidth={true}
                position={position[count]}
                open={count !== 0}
                offset={[0, 4]}
            >
                {popoverChildren}
            </Popover>

            <Button size='s' ref={buttonElementRef} onClick={onClick}>
                {label[count]}
            </Button>
        </div>
    );
});
```
