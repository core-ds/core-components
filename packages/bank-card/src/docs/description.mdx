Компонент в виде карты для ввода номера карты или банковского счета.

```jsx live
render(() => {
    const [value, setValue] = React.useState('');

    const handleChange = (event, payload) => setValue(payload.value);

    return (
        <>
            <BankCard value={value} onChange={handleChange} />
            <Gap size='s' />
            <Button size='xs' onClick={() => setValue('4111111111111111')}>
                Вставить номер карты
            </Button>
            <Gap size='s' direction='horizontal' />
            <Button size='xs' onClick={() => setValue('')}>
                Очистить
            </Button>
        </>
    );
});
```
