# Приведённый код и live-демо имеют демонстрационный характер и не являются финальной реализацией

```jsx live mobileHeight={560}
render(() => {
    const [step, setStep] = React.useState(1);

    const handleClick = () => {
        setStep((prev) => {
            if (prev === 5) {
                return 1;
            }
            return prev + 1;
        });
    };

    return (
        <Space size={24}>
            <Steps
                activeStep={step}
                defaultActiveStep={1}
                ordered={true}
                isVerticalAlign={true}
                interactive={false}
                animateSpring={true}
            >
                {[1, 2, 3, 4].map((item, key) => (
                    <PureCell key={key} verticalPadding='none' horizontalPadding='none'>
                        <PureCell.Content>
                            <PureCell.Main>
                                <PureCell.Text titleColor='primary' view='component-primary'>
                                    Шаг {item}
                                </PureCell.Text>
                            </PureCell.Main>
                        </PureCell.Content>
                    </PureCell>
                ))}
            </Steps>
            <Button view='secondary' onClick={() => handleClick()}>
                Далее
            </Button>
        </Space>
    );
});
```
