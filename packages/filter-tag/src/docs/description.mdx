## Группа фильтров

Обычно фильтры объединяются в группу, в рамках группы рекомендуется использовать один паттерн отображения лейбла и контролов. У выбранного тега может отображаться шеврон, крестик или и то и другое. Лейбл в выбранном состоянии можно скрывать для экономии места. Но если выбрано несколько значений, лейбл нужно оставить.

```jsx live
// Одиночный выбор
const CustomField = ({
    hideFieldLabel,
    selected,
    setSelectedItems,
    innerProps: { ref, ...restInnerProps },
    ...restProps
}) => {
    const checkedContent = (
        <span>
            {!hideFieldLabel && 'Одиночный выбор:'} {selected && selected.content}
        </span>
    );

    const contentLabel = <span>Одиночный выбор</span>;

    return (
        <div ref={ref}>
            <FilterTagDesktop
                onClear={() => setSelectedItems([])}
                view='filled'
                checked={selected}
                {...restInnerProps}
                {...restProps}
            >
                {selected ? checkedContent : contentLabel}
            </FilterTagDesktop>
        </div>
    );
};

// Множественный выбор
const CustomFieldMultiple = ({
    hideFieldLabel,
    selected,
    setSelectedItems,
    selectedMultiple,
    innerProps: { ref, ...restInnerProps },
    ...restProps
}) => {
    const content = selected && selected.content;

    const checkedContent =
        selectedMultiple.length > 1 ? (
            <span>Множественный выбор ({selectedMultiple.length})</span>
        ) : (
            <span>
                {!hideFieldLabel && 'Множественный выбор:'} {content}
            </span>
        );

    const contentLabel = <span>Множественный выбор</span>;

    return (
        <div ref={ref}>
            <FilterTagDesktop
                onClear={() => setSelectedItems([])}
                view='filled'
                checked={selected}
                {...restInnerProps}
                {...restProps}
            >
                {selected ? checkedContent : contentLabel}
            </FilterTagDesktop>
        </div>
    );
};

render(() => {
    const FIELD_SIZE = 40;
    const FIELD_SHAPE = 'rectangular';

    const controlMap = {
        chevron: { showArrow: true, showClear: false },
        clear: { showArrow: false, showClear: true },
        both: { showArrow: true, showClear: true },
    };

    const options = React.useMemo(
        () => [
            { key: '1', content: 'Aurum' },
            { key: '2', content: 'Bercelium' },
            { key: '3', content: 'Curium' },
            { key: '4', content: 'Neptunium' },
            { key: '5', content: 'Plutonuim' },
        ],
        [],
    );

    const [controls, setControls] = React.useState('chevron');
    const [hideLabel, setHideLabel] = React.useState(false);
    const { showArrow, showClear } = controlMap[controls];

    const fieldProps = {
        size: FIELD_SIZE,
        shape: FIELD_SHAPE,
        showArrow,
        showClear,
        hideFieldLabel: hideLabel,
    };

    const [selected, setSelected] = React.useState([]);

    const handleChangeSelected = ({ selectedMultiple }) => {
        setSelected(selectedMultiple);
    };

    const [selectedMultiple, setSelectedMultiple] = React.useState([]);
    const handleChangeMultiple = ({ selectedMultiple }) => {
        setSelectedMultiple(selectedMultiple.map((option) => option.key));
    };

    // Диапазон значений
    const [open, setOpen] = React.useState(false);
    const { selectedFrom, selectedTo, updatePeriod, resetPeriod } = usePeriod();
    const calendarRef = React.useRef(null);
    const anchorRef = React.useRef(null);

    const handleOpen = () => {
        setOpen(true);
    };

    const handleClose = () => {
        setOpen(false);
    };

    const handleClear = () => {
        handleClose();
        resetPeriod();
    };

    const handleUpdatePeriod = (date) => {
        updatePeriod(date);

        if (selectedFrom) {
            handleClose();
        }
    };

    const handleWrapperBlur = (event) => {
        const target = event.relatedTarget || document.activeElement;

        if (
            (anchorRef.current && anchorRef.current.contains(target)) ||
            (calendarRef.current && calendarRef.current.contains(target))
        ) {
            return;
        }

        handleClose();
    };

    const getDateString = React.useCallback((date) => {
        const day = date.getDate() < 10 ? `0${date.getDate()}` : date.getDate();
        const month = date.getMonth() + 1 < 10 ? `0${date.getMonth() + 1}` : date.getMonth() + 1;
        const year = date.getFullYear();

        return `${day}.${month}.${year}`;
    }, []);

    const selectedRange = React.useMemo(() => {
        if (selectedFrom && selectedTo) {
            const selectedFromDate = new Date(selectedFrom);
            const selectedToDate = new Date(selectedTo);
            return `${getDateString(selectedFromDate)} – ${getDateString(selectedToDate)}`;
        }
        return '';
    }, [selectedFrom, selectedTo]);

    const checkedContent = (
        <span>
            {!hideLabel && 'Диапазон значений:'} {selectedRange && selectedRange}
        </span>
    );
    const content = <span>Диапазон значений</span>;

    return (
        <>
            <style>
                {`
                    .filter-tag-options-list-desktop {
                        width: 200px;
                        min-width: 200px !important;
                    }
                    .filter-tag-calendar-popover {
                        border-radius: 0;
                        border: none;
                        background: transparent;
                        box-shadow: none;
                    }
                    .filter-tag-calendar-panel {
                        border-radius: var(--OptionListCornerRadius, 16px);
                        border: 1px solid var(--neutral-300, #E7E8EB);
                        background: var(--modal-bg-primary, #FFF);
                        box-shadow:
                            0 20px 24px 0 rgba(0, 0, 0, 0.08),
                            0 12px 16px 0 rgba(0, 0, 0, 0.04),
                            0 4px 8px 0 rgba(0, 0, 0, 0.04),
                            0 0 1px 0 rgba(0, 0, 0, 0.04);
                        overflow: hidden;
                    }
                `}
            </style>
            <SelectDesktop
                allowUnselect={true}
                popoverPosition='bottom-start'
                options={options}
                Field={CustomField}
                onChange={handleChangeSelected}
                selected={selected}
                Option={BaseOption}
                fieldProps={fieldProps}
                optionsListClassName='filter-tag-options-list-desktop'
            />
            <Gap size={16} />
            <SelectDesktop
                allowUnselect={true}
                popoverPosition='bottom-start'
                options={options}
                Field={CustomFieldMultiple}
                onChange={handleChangeMultiple}
                selected={selectedMultiple}
                multiple={true}
                fieldProps={fieldProps}
                optionsListClassName='filter-tag-options-list-desktop'
            />
            <Gap size={16} />
            <div ref={anchorRef} tabIndex={-1} onBlur={handleWrapperBlur}>
                <FilterTagDesktop
                    size={FIELD_SIZE}
                    shape={FIELD_SHAPE}
                    view='filled'
                    showArrow={showArrow}
                    showClear={showClear}
                    onClick={handleOpen}
                    onClear={handleClear}
                    checked={Boolean(selectedRange)}
                    open={open}
                >
                    {selectedRange ? checkedContent : content}
                </FilterTagDesktop>
                <Popover
                    anchorElement={anchorRef.current}
                    offset={[0, 4]}
                    open={open}
                    preventFlip={true}
                    position='bottom-start'
                    withTransition={false}
                    popperClassName='filter-tag-calendar-popover'
                >
                    <div
                        className='filter-tag-calendar-panel'
                        onMouseDown={(event) => event.preventDefault()}
                    >
                        <CalendarDesktop
                            ref={calendarRef}
                            responsive={true}
                            selectedFrom={selectedFrom}
                            selectedTo={selectedTo}
                            onChange={handleUpdatePeriod}
                            selectorView='month-only'
                        />
                    </div>
                </Popover>
            </div>

            <Gap size={24} />

            <RadioGroup
                direction='vertical'
                label='Контролы'
                name='filter-controls'
                onChange={(_, payload) => setControls(payload.value)}
                value={controls}
            >
                <Radio label='Шеврон' value='chevron' />
                <Radio label='Крестик' value='clear' />
                <Radio label='Шеврон и крестик' value='both' />
            </RadioGroup>

            <Gap size={16} />

            <RadioGroup
                direction='vertical'
                label='Лейбл в выбранном состоянии'
                name='filter-label'
                onChange={(_, payload) => setHideLabel(payload.value === 'hide')}
                value={hideLabel ? 'hide' : 'show'}
            >
                <Radio label='Показывать' value='show' />
                <Radio label='Скрывать' value='hide' />
            </RadioGroup>
        </>
    );
});
```

## Предзаполненное значение

Компонент может иметь предзаполненное значение, в таком случае, в заполненном состоянии у него не может быть крестика очистки.

```jsx live
const CustomField = ({
    label,
    selected,
    setSelectedItems,
    innerProps: { ref, ...restInnerProps },
    ...restProps
}) => (
    <div ref={ref}>
        <FilterTagDesktop checked={selected} showClear={false} {...restInnerProps} {...restProps}>
            {selected.content}
        </FilterTagDesktop>
    </div>
);
render(() => {
    const options = React.useMemo(

            { key: '1', content: 'По возрастанию' },

        [],

    const [selected, setSelected] = React.useState([options[0]]);

    const handleChangeSelected = ({ selectedMultiple }) => {
        setSelected(selectedMultiple);
    };

    return (
        <>
            <style>
                {`
                    .filter-tag-options-list-desktop {
                        width: 200px;
                        min-width: 200px !important;
                    }
                `}
            </style>
            <SelectDesktop
                popoverPosition='bottom-start'
                options={options}
                Field={CustomField}
```

## Состояния

Взаимодействие с тегом можно ограничить с помощью свойства [disabled]. Блокировать теги с выбранным значением не рекомендуется.

```jsx live
    const checkedContent = <span>Label: Value</span>;

                <FilterTagDesktop
                    size={40}
                    view='filled'
                    shape='rectangular'
                    checked={false}
                    showClear={false}
                    disabled={disabled}
                >
                    <span>Label</span>
                </FilterTagDesktop>
                <Gap size={12} direction='horizontal' />
                <FilterTagDesktop
                    size={40}
                    view='filled'

                    showClear={false}
                    disabled={disabled}
                >
                    {checkedContent}
                </FilterTagDesktop>

                <Gap size={12} direction='horizontal' />

                <FilterTagDesktop
                    size={40}
                    view='filled'
                    shape='rectangular'
                    checked={true}
                    showArrow={false}
                    disabled={disabled}

                    {checkedContent}
                </FilterTagDesktop>

                <Gap size={12} direction='horizontal' />

                <FilterTagDesktop
```

## Стиль

Компонент наследует опции компонента [Tag](?path=/docs/tag--docs), отвечающие за внешний вид:

- shape (форма) — овальные или прямоугольные.
- view (стиль) — залитые или бордерные.
