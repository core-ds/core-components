## Пример

Компоненту можно передать длину кода, в этом случае он сразу отобразит нужное количество точек над пин-падом.
Если длина неизвестна, после каждого нажатия на пин-пад будет добавляться новая точка.
Кнопка удаления появляется после ввода первого символа.
После валидации компонент может отобразить состоянии ошибки.
Корректная комбинация для примера `0451`.

```jsx live mobileHeight={640}
const VALID_CODE = '0451';
const MAX_LENGTH = VALID_CODE.length;
const SCREENS = {
    INITIAL: 'initial',
    ENTER_CODE: 'ENTER_CODE',
    SUCCESS: 'success',
};

render(() => {
    const [screen, setScreen] = React.useState(SCREENS.INITIAL);
    const [error, setError] = React.useState(false);
    const [errorMessage, setErrorMessage] = React.useState(false);
    const [code, setCode] = React.useState('');
    const [passCodeParams, setPassCodeParams] = React.useState();
    const passCodeRef = React.useRef(null);
    const containerRef = React.useRef(null);
    const [isSuccess, setSuccess] = React.useState(false);

    const validate = (codeToValidate) => {
        if (codeToValidate !== VALID_CODE) {
            setTimeout(() => {
                setError(true)
                setErrorMessage(true);
            }, 300);
            setTimeout(() => {
                if ('codeLength' in passCodeParams) {
                    setError(false);
                }
                setCode('');
                setErrorMessage(false);
            }, 1300);

            return;
        }
        setTimeout(() => {
            setSuccess(true);
        }, 700);

        setTimeout(() => {
            if ('codeLength' in passCodeParams) {
                setSuccess(false);
            }
            setCode('');
        }, 1300);

        setTimeout(() => setScreen(SCREENS.SUCCESS), 2100);
    };

    const handleChange = (value) => {
        setCode(value);
        setError(false);
        setSuccess(false);
        setErrorMessage(false);

        if (passCodeParams.codeLength > 0 && value.length === MAX_LENGTH) {
            validate(value);
        }
    };

    const renderMessage = () => {
        if ('codeLength' in passCodeParams) {
            return (
                <div style={{ textAlign: 'center' }}>
                    <Typography.Text view='primary-medium'>
                        Введите код из четырёх цифр
                    </Typography.Text>
                </div>
            );
        }

        return null;
    };

    if (screen === SCREENS.INITIAL) {
        return (
            <div
                style={{
                    height: 600,
                    display: 'flex',
                    flexDirection: 'column',
                    justifyContent: isMobile() ? 'start' : 'center',
                    alignItems: 'center',
                }}
            >
                <div style={{ width: 320 }}>
                    <Button
                        block
                        onClick={() => {
                            setPassCodeParams({
                                codeLength: MAX_LENGTH,
                                message: 'Введите код из четырёх цифр',
                            });
                            setScreen(SCREENS.ENTER_CODE);
                        }}
                    >
                        Код из четырех цифр
                    </Button>
                    <Gap size='m' />
                    <Button
                        block
                        onClick={() => {
                            setPassCodeParams({ maxCodeLength: 12 });
                            setScreen(SCREENS.ENTER_CODE);
                        }}
                    >
                        Длина кода неизвестна заранее
                    </Button>
                </div>
            </div>
        );
    }

    if (screen === SCREENS.SUCCESS) {
        return (
            <div style={{ display: 'flex', flexFlow: 'column nowrap', height: 600 }}>
                <Gap size='xl' />
                <Gap size='8xl' />
                <div
                    style={{
                        display: 'flex',
                        flexFlow: 'column nowrap',
                        alignItems: 'center',
                        justifyContent: 'space-between',
                        ...(isMobile() && { flex: 1 }),
                    }}
                >
                    <div style={{ textAlign: 'center', width: isMobile() ? 315 : 472 }}>
                        <SuperEllipse
                            size={80}
                            backgroundColor='var(--color-light-graphic-positive)'
                        >
                            <CheckmarkMIcon style={{ fill: '#fff' }} />
                        </SuperEllipse>

                        <Gap size='m' />

                        <Typography.Title view='small' font='system'>
                            Введён корректный пароль
                        </Typography.Title>

                        <Gap size={16} />

                        <Typography.Text view='primary-medium'>
                            Это пример экрана, на который попадает пользователь после ввода
                            корректного пароля
                        </Typography.Text>
                    </div>

                    <Gap size={24} />

                    <Button
                        onClick={() => {
                            setScreen(SCREENS.INITIAL);
                            setError(false);
                            setSuccess(false);
                            setCode('');
                        }}
                        style={{
                            ...(isMobile() && { width: 320 }),
                        }}
                    >
                        Попробовать ещё раз
                    </Button>
                </div>
            </div>
        );
    }

    if (isMobile()) {
        return (
            <div
                style={{
                    height: 600,
                    boxSizing: 'border-box',
                    paddingTop: passCodeParams.maxCodeLength > 0 ? '60px' : '34px',
                }}
                ref={containerRef}
            >
                <div style={{ height: passCodeParams.maxCodeLength > 0 ? 464 : 504 }}>
                    <Toast
                        title={'Error message'}
                        open={Boolean(errorMessage)}
                        anchorElement={containerRef.current}
                        position='top'
                        badge='negative-alert'
                        offset={[0, -56]}
                        style={{ width: '320px' }}
                        onClose={() => {}}
                    />
                    {renderMessage()}
                    <PassCode
                        {...passCodeParams}
                        ref={passCodeRef}
                        value={code}
                        onChange={handleChange}
                        error={error}
                        success={isSuccess}
                        leftAddons={
                            <KeyPadButton view='text' key='left-addon'>
                                Забыли код?
                            </KeyPadButton>
                        }
                        rightAddons={
                            <KeyPadButton view='text' key='right-addon'>
                                <SfFaceIdXxlIcon />
                            </KeyPadButton>
                        }
                    />

                    {passCodeParams.maxCodeLength > 0 && (
                        <React.Fragment>
                            <Gap size={20} />
                            <Button
                                style={{ display: 'block', width: '304px', margin: '0 auto' }}
                                view='primary'
                                onClick={() => validate(code)}
                            >
                                Продолжить
                            </Button>
                        </React.Fragment>
                    )}
                </div>
            </div>
        );
    }

    return (
        <div
            style={{
                height: 600,
                boxSizing: 'border-box',
                paddingTop: '107px',
            }}
            ref={containerRef}
        >
            <Toast
                title={'Error message'}
                open={Boolean(errorMessage)}
                anchorElement={containerRef.current}
                position='top'
                badge='negative-alert'
                offset={[0, -56]}
                style={{ width: '304px' }}
                onClose={() => {}}
            />
            {renderMessage()}
            <PassCode
                {...passCodeParams}
                ref={passCodeRef}
                value={code}
                onChange={handleChange}
                error={error}
                success={isSuccess}
                leftAddons={
                    <KeyPadButton view='text' key='left-addon'>
                        Забыли код?
                    </KeyPadButton>
                }
                rightAddons={
                    <KeyPadButton view='text' key='right-addon'>
                        <SfFaceIdXxlIcon />
                    </KeyPadButton>
                }
            />

            {passCodeParams.maxCodeLength > 0 && (
                <Button
                    style={{ display: 'block', width: '304px', margin: '0 auto' }}
                    view='primary'
                    onClick={() => validate(code)}
                >
                    Продолжить
                </Button>
            )}
        </div>
    );
});
```
