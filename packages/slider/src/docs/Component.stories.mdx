import { Meta, Story, Markdown } from '@storybook/addon-docs';
import { number, select, boolean, object } from '@storybook/addon-knobs';
import { ComponentHeader, Tabs } from 'storybook/blocks';
import { Slider } from '@alfalab/core-components-slider';

import Description from './description.mdx';
import Development from './development.mdx';
import Changelog from '../../CHANGELOG.md?raw';

<Meta title='Components/Slider' component={Slider} id='Slider' />

{/* Canvas */}

<Story name='Slider'>
    {React.createElement(() => {
        const [value, setValue] = React.useState(50);
        const handleChange = ({ value }) => setValue(value);
        return (
            <Slider
                value={value}
                valueTo={number('valueTo', 0)}
                onChange={handleChange}
                min={number('min', 0)}
                max={number('max', 100)}
                step={number('step', 1)}
                size={select('size', [2, 4], 4)}
                behaviour={select('behaviour', ['unconstrained-tap', 'tap'], 'tap')}
                dots={boolean('dots', true)}
                pipsLabel={select('pipsLabel', ['all', 'pipsOnly', 'customPipsOnly', 'none'], 'all')}
                dotsSlider={select('dotsSlider', ['step', 'custom'], 'step')}
                showPipsDots={boolean('showPipsDots', false)}
                customDots={object('customDots', [25, 50, 75])}
            />
        );
    })}
</Story>

{/* Docs */}

<ComponentHeader
    name='Slider'
    children='Компонент используется для указания значения из заданного диапазона.'
/>

<Tabs
    description={<Description />}
    development={<Development />}
    changelog={<Markdown>{Changelog}</Markdown>}
/>
