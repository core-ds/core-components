Плашка для компонента `Toast`. Также используется в компоненте `Notification`.

```jsx live
render(() => {
    const isMobile = document.body.clientWidth < 450;
    return (
        <ToastPlate
            badge='positive'
            title='Поздравляем, полный успех'
            hasCloser={true}
            block={true}
            bottomButtonPosition={isMobile}
            onClose={Function.prototype}
            actionButton={
                <Button colors='inverted' view='ghost' size={isMobile ? 'xs' : 's'}>
                    Восстановить
                </Button>
            }
        />
    );
});
```
