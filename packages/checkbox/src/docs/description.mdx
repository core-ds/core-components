## Размеры

Компонент поддерживате два размера, 24 и 20 px

```jsx live
render(() => {
   const [checked, setChecked] = React.useState({
        checkbox24: true,
        checkbox20: true,
    });

   const [switchChecked, setSwitchChecked] = React.useState(true);
   const [error, setError] = React.useState(false);
   const [disabled, setDisabled] = React.useState(false);

    const handleChange = (name) => (event) => {
        setChecked((prevState) => ({
            ...prevState,
            [name]: event.target.checked,
        }));
    };


    return (
        <React.Fragment>
            <div style={{ width: document.body.clientWidth < 450 ? '100%' : 320, minHeight: '108px', marginBottom: '20px' }}>
                <div style={{minHeight: '46px'}}>
                    <Checkbox
                        block={true}
                        size={24}
                        onChange={handleChange('checkbox24')}
                        checked={checked.checkbox24}
                        disabled={disabled}
                        label='Согласен с условиями'
                        {...(switchChecked && {hint: 'И внимательно их прочитал'})}
                        {...(error && {error: 'Error message'})}
                    />
                </div>
                <div style={{margin: '20px 0 0 8px'}}>
                    <Checkbox
                        block={true}
                        size={20}
                        onChange={handleChange('checkbox20')}
                        checked={checked.checkbox20}
                        disabled={disabled}
                        label='Согласен с условиями'
                        {...(switchChecked && {hint: 'И внимательно их прочитал'})}
                        {...(error && {error: 'Error message'})}
                    />
                </div>
            </div>
            <Divider />
            <div style={{margin: '20px 0'}}>
                <Switch
                    block={true}
                    checked={switchChecked}
                    label='Подпись под лейблом'
                    onChange={() => setSwitchChecked((prevState) => !prevState)}
                />
            </div>

            <Button view="secondary" size="32"  onClick={() => setError((prevState) => !prevState)}>Ошибка</Button>
            <span style={{marginLeft: '8px'}}>
                <Button view="secondary" size="32" onClick={() => setDisabled((prevState) => !prevState)}>Заблокирован</Button>
            </span>
        </React.Fragment>
    );
});
```

## Состояния

Чекбокс может находиться в выбранном и невыбранном состоянии. В редких кейсах его значение может быть неопределённым.
Если чекбокс заблокирован, используется состояние «недоступен».

```jsx live
render(() => {
    return (
        <div style={{ width: document.body.clientWidth < 450 ? '100%' : 320 }}>
            <Checkbox size={24} label='Неопределён' indeterminate={true} block={true} />
            <Gap size='xl' />
            <Checkbox size={24} label='Выбран' checked={true} block={true} />
            <Gap size='m' />
            <Checkbox
                size={24}
                label='Выбран и недоступен'
                disabled={true}
                checked={true}
                block={true}
            />
            <Gap size='m' />
            <Checkbox size={24} label='По умолчанию' checked={false} block={true} />
            <Gap size='m' />
            <Checkbox
                size={24}
                label='По умолчанию и недоступен'
                checked={false}
                disabled={true}
                block={true}
            />
        </div>
    );
});
```
