import { Meta, Story, Markdown } from '@storybook/addon-docs';
import { text, boolean, select } from '@storybook/addon-knobs';
import { ComponentHeader, Tabs } from 'storybook/blocks';
import { Checkbox } from '@alfalab/core-components-checkbox';

import packageJson from '../../package.json';
import Description from './description.mdx';
import Development from './development.mdx';
import Changelog from '../../CHANGELOG.md?raw';

<Meta title='Components/Checkbox' component={Checkbox} id='Checkbox' />

{/* Canvas */}

<Story name='Checkbox'>
    {React.createElement(() => {
        const [checked, setChecked] = React.useState(false);
        const handleChange = () => setChecked(!checked);
        const size = select('size', ['s', 'm'], 's');
        const disabled = boolean('disabled', false);
        const inactive = boolean('inactive', false);
        const checkedKnob = boolean('checked', false);
        const label = text('label', 'Согласен с условиями');
        const indeterminate = boolean('indeterminate', false);
        const hint = text('hint', 'Дополнительная информация');
        const error = text('error', '');
        const block = boolean('block', false);
        const addons = text('addons');
        const align = select('align', ['start', 'center']);
        return (
            <Checkbox
                disabled={disabled}
                inactive={inactive}
                size={size}
                block={block}
                addons={addons}
                align={align}
                onChange={handleChange}
                checked={checked || checkedKnob}
                label={label}
                indeterminate={indeterminate}
                hint={hint}
                error={error}
            />
        );
    })}
</Story>

{/* Docs */}

<ComponentHeader
    name='Checkbox'
    version={packageJson.version}
    package='@alfalab/core-components/checkbox'
    design='https://www.figma.com/file/cdNnkh2QdxuvYLrBm4cubM/Web-%3A%3A-Core-Default-Components?type=design&node-id=87496%3A109864&mode=design&t=95S9JXl5nVy6yP2f-1'
    children='Используется для управления параметром с бинарным состоянием.'
/>

<Tabs
    description={<Description />}
    development={<Development />}
    changelog={<Markdown>{Changelog}</Markdown>}
/>
