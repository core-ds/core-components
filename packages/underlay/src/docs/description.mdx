## Примеры

Настраиваются: цвет фона, тень, радиус скругления, величина отступов, цвет и толщина обводки. 
При необходимости, можно сделать компонент кликабельным.

```jsx live
render(() => {
    const styleContent = {
        width: '288px',
        height: '168px',
        borderRadius: '8px',
        backgroundColor: 'rgba(55, 120, 251, 0.1)',
        border: '1px dashed rgb(55, 120, 251)',
        boxSizing: 'border-box',
    }
    
    const handleClick = () => {
      alert('Клик!')
    } 

    const css = `
        div[data-test-id='test1'] {
            cursor: pointer;
            width: 320px;
            height: 200px;
        }
        
        div[data-test-id='test2'] {
            background-image: url(./images/underlay.png);
            background-repeat: no-repeat;            
        }
    `;

    return (
    <Container>
        <Underlay 
            onClick={handleClick}
            borderSize={1}
            borderRadius='xl'
            padding={{
                top: 'm',
                left:'m',
                bottom: 'm',
                right: 'm'
            }}
            backgroundColor='positive-muted'
            borderColor='graphic-positive'
            dataTestId='test1'
        >
            <div style={styleContent} />
        </Underlay>
        <style>{css}</style>
        <Gap size='l'/>
        <Underlay 
            padding='m'
            dataTestId='test2'
            >
                <div style={styleContent}/>
        </Underlay>
    </Container>
    );
});
```
