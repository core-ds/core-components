### PureCell
Универсальный конструктор для сборки любой ячейки

```tsx live
render (() => {
    const padding = select('padding', ['airy', 'default', 'compact', 'tiny'], 'default');
    const direction = select('direction', ['vertical', 'horizontal'], 'horizontal');

    return (
        <PureCell href={ text('href', '') } padding={ padding } direction={ direction }>
            <PureCell.Graphics>
                <Circle />
            </PureCell.Graphics>
            <PureCell.Content>
                <PureCell.Main>
                    <PureCell.Text color="primary" view="component">
                        Title
                    </PureCell.Text>
                    <PureCell.Text color="secondary" view="primary-small">
                        Label
                    </PureCell.Text>
                </PureCell.Main>
            </PureCell.Content>
        </PureCell>
    );
});
```
