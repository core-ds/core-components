## Примеры

```jsx live
render(() => {
    const getChildStyle = (bgColor) => ({
        height: 80,
        textAlign: 'center',
        lineHeight: '80px',
        fontSize: 40,
        fontWeight: 700,
        color: 'var(--color-light-text-tertiary)',
        background: `var(${bgColor})`,
    });
    return (
        <div style={{ height: 240, margin: 'var(--gap-l-neg)' }}>
            <Scrollbar style={{ height: '100%' }} autoHide={false}>
                {new Array(51).fill(null).map((_, idx) => {
                    const style = getChildStyle(
                        idx % 2 === 0 ? '--color-light-bg-tertiary' : '--color-light-bg-secondary',
                    );
                    return (
                        <div key={idx} style={style}>
                            {idx + 1}
                        </div>
                    );
                })}
            </Scrollbar>
        </div>
    );
});
```

render(() => {

    return (
        <Scrollbar style={scrollbarStyle} autoHide={false}>

                const style = getChildStyle(
                    idx % 2 === 0 ? '--color-light-bg-tertiary' : '--color-light-bg-secondary',
                );
                return (
                    <div key={idx} style={style}>
                        {idx + 1}
                    </div>
                );
            })}
        </Scrollbar>
    );
});
