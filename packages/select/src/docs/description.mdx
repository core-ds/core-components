## Пример

```jsx live
const OPTIONS = [
    { key: '1', content: 'Neptunium' },
    { key: '2', content: 'Plutonium' },
    { key: '3', content: 'Americium' },
    { key: '4', content: 'Curium' },
    { key: '5', content: 'Berkelium' },
    { key: '6', content: 'Californium' },
    { key: '7', content: 'Einsteinium' },
    { key: '8', content: 'Fermium' },
];

render(() => {
    const [selectedFirst, setSelectedFirst] = React.useState([]);
    const ref = React.useRef(null);
    const [virtual, setVirtual] = React.useState(false);
    const [minHeight, setMinHeight] = React.useState(false);

    return (
        <div style={{ width: 320 }}>
            <style>{`.foo {display: flex;} .bar {overflow-y: unset; display: flex;} .far {overflow-y: unset; display: flex; min-height: 230px;} .buz { display: flex; flex-direction: column; max-height: 100%;} .fuz {display: flex;flex-direction: column;}`}</style>
            <Switch
                checked={virtual}
                onChange={() => setVirtual((prev) => !prev)}
                label='VirtualOptionsList'
            />
            <Gap size='m' />
            <Switch
                checked={minHeight}
                onChange={() => setMinHeight((prev) => !prev)}
                label='min-height'
            />
            <Gap size='m' />
            <SelectDesktop
                allowUnselect={true}
                size={56}
                placeholder='Выберите элементы'
                label='Множественный выбор'
                Option={BaseOption}
                multiple={true}
                block={true}
                {...useSelectWithApply({
                    options: OPTIONS,
                    selected: selectedFirst,
                    onChange: ({ selectedMultiple }) => {
                        setSelectedFirst(selectedMultiple.map((option) => option.key));
                    },
                    // classNames
                    optionsListProps: { className: 'buz', scrollbarClassName: 'fuz' },
                    OptionsList: virtual ? VirtualOptionsList : OptionsList,
                })}
                // className
                popperClassName='foo'
                // className
                popoverProps={{
                    scrollableContentClassName: minHeight ? 'far' : 'bar',
                    // availableHeight
                    availableHeight: () => {
                        return ref.current;
                    },
                }}
            />
            <Gap size='m' />
            <textarea ref={ref} placeholder='Resize me then click Select' />
        </div>
    );
});
```
