import { Meta, Story, Markdown } from '@storybook/addon-docs';
import { text, boolean, select } from '@storybook/addon-knobs';
import { ComponentHeader, Tabs } from 'storybook/blocks';
import { createComponentUrl } from 'storybook/utils/createComponentUrl';
import { Radio } from '@alfalab/core-components-radio';
import { Link } from '@alfalab/core-components-link';

import Description from './description.mdx';
import Development from './development.mdx';
import Changelog from '../../CHANGELOG.md?raw';

<Meta title='Components/Radio' component={Radio} id='Radio' />

{/* Canvas */}

<Story name='Radio'>
    {React.createElement(() => {
        const [checked, setChecked] = React.useState(false);
        const handleChange = () => setChecked(!checked);
        const size = select('size', [20, 24], 20);
        const disabled = boolean('disabled', false);
        const inactive = boolean('inactive', false);
        const checkedKnob = boolean('checked', false);
        const block = boolean('block', false);
        const label = text('label', 'Text label');
        const hint = text('hint', 'Hint');
        const align = select('align', ['start', 'center'], 'start');
        const addons = text('addons');
        const colors = select('colors', ['default', 'inverted'], 'default');
        return (
            <div
                style={{
                    backgroundColor:
                        colors === 'inverted'
                            ? 'var(--color-light-base-bg-primary-inverted)'
                            : 'transparent',
                    padding: '8px',
                    position: 'absolute',
                    top: 0,
                    left: 0,
                    right: 0,
                    bottom: 0,
                }}
            >
                <Radio
                    disabled={disabled}
                    inactive={inactive}
                    addons={addons}
                    block={block}
                    size={size}
                    align={align}
                    onChange={handleChange}
                    checked={checked || checkedKnob}
                    hint={hint}
                    label={label}
                    colors={colors}
                />
            </div>
        );
    })}
</Story>

{/* Docs */}

<ComponentHeader name='Radio'>
    {'Используется только в составе\u00A0'}
    <Link view='default' underline={false} href={createComponentUrl('RadioGroup')}>
        {'RadioGroup.'}
    </Link>
</ComponentHeader>

<Tabs
    description={<Description />}
    development={<Development />}
    changelog={<Markdown>{Changelog}</Markdown>}
/>
