<Story name='SuperEllipse'>
    {React.createElement(() => {
        const topAddons = boolean('topAddons', false);
        const bottomAddons = boolean('bottomAddons', false);
        const indicator = boolean('indicator', false);
        const size = select('size', ['20', '24', '32', '40', '48', '64', '80', '128'], '64');
        const sizeAddons = size === '128' ? '40' : size === '80' || size === '64' ? '24' : '20';
        const addonsIcon = <Circle size={sizeAddons}> {size === '20' ? <DiamondsSIcon/> : <DiamondsMIcon/>} </Circle>;
        return (
            <SuperEllipse
                size={size}
                backgroundColor={text('backgroundColor', '#f3f4f5')}
                border={boolean('border', false)}
                imageUrl={text('imageUrl', '')}
                topAddons={topAddons && addonsIcon}
                bottomAddons={bottomAddons && addonsIcon}
                indicator={indicator && <Badge view='count' />}
            >
                { size === '128' || size === '80' ?  <DiamondsXxlIcon/> : size === '20' ? <DiamondsSIcon/> : <DiamondsMIcon/>}
            </SuperEllipse>
        );
    })}
</Story>
